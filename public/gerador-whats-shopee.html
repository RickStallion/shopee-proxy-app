import fetch from 'node-fetch'; import { sendCors, requireAuth } from './_cors.js'; const UA = 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122 Safari/537.36'; function setNoCache(res) { res.setHeader('Cache-Control', 'no-store, no-cache, must-revalidate'); res.setHeader('Pragma', 'no-cache'); res.setHeader('Expires', '0'); res.setHeader('CDN-Cache-Control', 'no-store'); res.setHeader('Vercel-CDN-Cache-Control', 'no-store'); } function parseIds(u) { try { const url = new URL(u); const m = url.pathname.match(/(?:^|\/)i\.(\d+)\.(\d+)(?:$|[/?])/i); if (m) return { shopid: m[1], itemid: m[2] }; } catch {} return { shopid: null, itemid: null }; } function brlFromShopeeInt(p) { if (p == null) return ''; const val = Number(p) / 100000; if (!isFinite(val)) return ''; return 'R$ ' + val.toFixed(2).replace('.', ','); } function toBRL(x) { if (x == null || x === '') return ''; let s = String(x).replace(/[^\d.,]/g, ''); if (!s) return ''; if (!/[.,]/.test(s)) s = s.replace(/(\d+)(\d{2})$/, '$1,$2'); return 'R$ ' + s; } async function fetchJson(url, hdr = {}) { const r = await fetch(url, { headers: { 'user-agent': UA, ...hdr }, redirect: 'follow' }).catch(() => null); if (!r || !r.ok) return null; const txt = await r.text().catch(() => ''); try { return JSON.parse(txt); } catch { return null; } } async function fetchReadable(url) { const thru = 'https://r.jina.ai/http://' + url.replace(/^https?:\/\//, ''); const r = await fetch(thru, { headers: { 'user-agent': UA, 'accept-language': 'pt-BR,pt;q=0.9' }, redirect: 'follow' }).catch(() => null); if (!r || !r.ok) return ''; return await r.text().catch(() => ''); } function extractFromReadable(text) { let title = ''; let image = ''; let price = ''; // título let m = text.match(/og:title["']?\s*content=["']([^"\n]{5,})/i) || text.match(/"name"\s*:\s*"([^"]{5,})"/i); if (m) title = m[1]; // preço const priceRX = [ /"price"\s*:\s*"?([\d.,]+)"?/i, /"current_price"\s*:\s*([\d.]+)/i, /"final_price"\s*:\s*([\d.]+)/i, /"price_min"\s*:\s*([\d.]+)/i, /"price_before_discount"\s*:\s*([\d.]+)/i, /R\$\s*([\d.,]+)/i ]; for (const rx of priceRX) { const pm = text.match(rx); if (pm && pm[1]) { price = pm[1]; break; } } // imagem m = text.match(/"image(?:s)?"\s*:\s*\[\s*"([^"]+)"/i) || text.match(/"image"\s*:\s*"([^"]+)"/i); if (m) image = m[1]; if (image && !/^https?:\/\//i.test(image) && image.length > 20) { image = `https://cf.shopee.com.br/file/${image}`; } return { title: (title || '').replace(/\s+/g, ' ').trim(), price: toBRL(price), image: (image || '').trim() }; } export default async function handler(req, res) { if (sendCors(req, res)) return; if (!requireAuth(req, res)) return; setNoCache(res); try { const u = new URL(req.url, 'http://x'); const target = u.searchParams.get('u'); if (!target) return res.status(400).json({ ok: false, error: 'missing u' }); // segue redirecionadores let finalUrl = target; try { const r = await fetch(target, { headers: { 'user-agent': UA }, redirect: 'follow' }); if (r && r.url) finalUrl = r.url; } catch {} // 1) Tenta API oficial let { shopid, itemid } = parseIds(finalUrl); if (shopid && itemid) { // direto let item = await (async () => { const j = await fetchJson(`https://shopee.com.br/api/v4/item/get?shopid=${shopid}&itemid=${itemid}`, { 'accept': 'application/json', 'accept-language': 'pt-BR,pt;q=0.9', 'referer': finalUrl }); if (j && (j.item || j.data)) return j.item || j.data; return null; })(); // via r.jina.ai (caso bloqueie) if (!item) { const thru = 'https://r.jina.ai/http://shopee.com.br/api/v4/item/get?shopid=' + shopid + '&itemid=' + itemid; const j = await fetchJson(thru, { 'accept': 'application/json' }); if (j && (j.item || j.data)) item = j.item || j.data; } if (item) { const title = item.name || ''; const imgHash = item.image || (Array.isArray(item.images) && item.images[0]) || ''; const image = imgHash ? `https://cf.shopee.com.br/file/${imgHash}` : ''; const raw = item.price_min ?? item.price ?? item.current_price ?? item.final_price; const price = raw != null ? brlFromShopeeInt(raw) : ''; return res.json({ ok: true, mode: 'api', finalUrl, title, image, price }); } } // 2) Fallback: página renderizada via r.jina.ai const readable = await fetchReadable(finalUrl); const { title, price, image } = extractFromReadable(readable); return res.json({ ok: true, mode: 'readable', finalUrl, title, image, price }); } catch (e) { return res.status(500).json({ ok: false, error: String(e?.message || e) }); } }
